    <script>

        // Elegant gold dust sparkle particles floating in background

        const canvas = document.getElementById('gold-dust-canvas');

        const ctx = canvas.getContext('2d');

        let particles = [];

        const maxParticles = 30;

        function resizeCanvas() {

            canvas.width = window.innerWidth;

            canvas.height = window.innerHeight;

        }

        window.addEventListener('resize', resizeCanvas);

        resizeCanvas();

        class Particle {

            constructor() {

                this.reset();

            }

            reset() {

                this.x = Math.random() * canvas.width;

                this.y = Math.random() * canvas.height;

                this.size = Math.random() * 1.5 + 0.5;

                this.speedX = Math.random() * 0.2 - 0.1;

                this.speedY = Math.random() * 0.3 + 0.05;

                this.alpha = Math.random() * 0.4 + 0.1;

                this.alphaSpeed = Math.random() * 0.005 + 0.001;

            }

            update() {

                this.x += this.speedX;

                this.y += this.speedY;

                this.alpha += this.alphaSpeed;

                if (this.alpha > 0.6 || this.alpha < 0.1) {

                    this.alphaSpeed = -this.alphaSpeed;

                }

                if (this.y > canvas.height) {

                    this.reset();

                    this.y = 0;

                }

            }

            draw() {

                ctx.save();

                ctx.globalAlpha = this.alpha;

                ctx.fillStyle = '#d4af37';

                ctx.beginPath();

                ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);

                ctx.fill();

                ctx.restore();

            }

        }

        for (let i = 0; i < maxParticles; i++) {

            particles.push(new Particle());

        }

        function animate() {

            ctx.clearRect(0, 0, canvas.width, canvas.height);

            particles.forEach(p => {

                p.update();

                p.draw();

            });

            requestAnimationFrame(animate);

        }

        animate();

        // Web Audio API Sound Generation (Harp-like sound)

        let audioCtx = null;

        let isMusicPlaying = false;

        let melodyTimer = null;

        function playOpeningChime() {

            try {

                const actx = new (window.AudioContext || window.webkitAudioContext)();

                const chimeFreqs = [523.25, 659.25, 783.99, 987.77, 1046.50]; // Beautiful C Maj9

                chimeFreqs.forEach((freq, i) => {

                    const osc = actx.createOscillator();

                    const gain = actx.createGain();

                    osc.type = 'sine';

                    osc.frequency.setValueAtTime(freq, actx.currentTime + (i * 0.1));

                    gain.gain.setValueAtTime(0, actx.currentTime + (i * 0.1));

                    gain.gain.linearRampToValueAtTime(0.06, actx.currentTime + (i * 0.1) + 0.05);

                    gain.gain.exponentialRampToValueAtTime(0.001, actx.currentTime + (i * 0.1) + 0.8);

                    osc.connect(gain);

                    gain.connect(actx.destination);

                    osc.start(actx.currentTime + (i * 0.1));

                    osc.stop(actx.currentTime + (i * 0.1) + 0.9);

                });

            } catch(e) {}

        }

        const toggleBtn = document.getElementById('music-toggle-btn');

        toggleBtn.addEventListener('click', () => {

            if (isMusicPlaying) {

                stopMusicSynth();

            } else {

                startMusicSynth();

            }

        function startMusicSynth() {

            if (isMusicPlaying) return;

            try {

                audioCtx = new (window.AudioContext || window.webkitAudioContext)();

                isMusicPlaying = true;

                toggleBtn.classList.remove('hidden');

                document.getElementById('music-wave').classList.remove('hidden');

                document.getElementById('music-icon').classList.add('text-yellow-600');

                // Soft background chords loop

                const chords = [

                    [261.63, 329.63, 392.00, 493.88], // Cmaj7

                    [293.66, 349.23, 440.00, 523.25], // Dm7

                    [220.00, 261.63, 329.63, 392.00], // Am7

                    [196.00, 246.94, 293.66, 392.00]  // G7

                ];

                let chordIndex = 0;

                function playMelody() {

                    if (!isMusicPlaying || !audioCtx) return;

                    const now = audioCtx.currentTime;

                    const chord = chords[chordIndex];

                    chord.forEach((freq, noteIdx) => {

                        const noteDelay = noteIdx * 0.25;

                        const osc = audioCtx.createOscillator();

                        const gain = audioCtx.createGain();

                        const lowpass = audioCtx.createBiquadFilter();

                        osc.type = 'triangle';

                        osc.frequency.setValueAtTime(freq, now + noteDelay);

                        lowpass.type = 'lowpass';

                        lowpass.frequency.setValueAtTime(800, now + noteDelay);

                        gain.gain.setValueAtTime(0, now + noteDelay);

                        gain.gain.linearRampToValueAtTime(0.04, now + noteDelay + 0.3);

                        gain.gain.exponentialRampToValueAtTime(0.001, now + noteDelay + 2.5);

                        osc.connect(lowpass);

                        lowpass.connect(gain);

                        gain.connect(audioCtx.destination);

                        osc.start(now + noteDelay);

                        osc.stop(now + noteDelay + 3.0);

                    });

                    chordIndex = (chordIndex + 1) % chords.length;

                }

                playMelody();

                melodyTimer = setInterval(playMelody, 4000);

            } catch(e) {}

        }

        function stopMusicSynth() {

            isMusicPlaying = false;

            if (melodyTimer) clearInterval(melodyTimer);

            if (audioCtx) {

                audioCtx.close();

                audioCtx = null;

            }

            document.getElementById('music-wave').classList.add('hidden');

            document.getElementById('music-icon').classList.remove('text-yellow-600');

        }
